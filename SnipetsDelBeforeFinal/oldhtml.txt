    const milconSlider = document.getElementById('input_milcon_slider');
    const oamSlider = document.getElementById('input_oam_slider');

    const updateSliderValue = (slider, outputId) => {
        const outputElement = slider.previousElementSibling.querySelector('span');
        slider.addEventListener('input', (event) => {
            outputElement.textContent = event.target.value;

            // Simulate updating the funding fill
            const fillElement = slider.previousElementSibling.querySelector('.funding-fill');
            fillElement.style.width = event.target.value + '%';
        });
    };

    if (milconSlider) {
        updateSliderValue(milconSlider, 'output_milcon_value');
    }
    if (oamSlider) {
        updateSliderValue(oamSlider, 'output_oam_value');
    }
</script>
